% XNOR gate - standard ANSI/IEEE distinctive-shape symbol.
% Like XOR but with a bubble on the output (bubble = inversion).
\documentclass[border=12pt]{standalone}
\usepackage{circuitikz}
\begin{document}
\begin{circuitikz}[line width=0.8pt, scale=1.4, transform shape]
  \node[xnor port] (g) {};
  \draw (g.in 1) -- ++(-1,0) node[left]{$A$};
  \draw (g.in 2) -- ++(-1,0) node[left]{$B$};
  \draw (g.out)  -- ++(1,0)  node[ocirc]{} node[right]{$Y=\overline{A\oplus B}$};
\end{circuitikz}
\end{document}
